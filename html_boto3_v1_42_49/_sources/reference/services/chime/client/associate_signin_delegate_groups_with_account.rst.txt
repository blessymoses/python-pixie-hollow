.. py:method:: Chime.Client.associate_signin_delegate_groups_with_account(**kwargs)

        AccountId='string',
        SigninDelegateGroups=[
            {
                'GroupName': 'string'
            },
        ]
    )
    
  :type AccountId: string
  :param AccountId: **[REQUIRED]**
