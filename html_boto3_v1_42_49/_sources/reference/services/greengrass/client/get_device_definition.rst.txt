  ::

    response = client.get_device_definition(
        DeviceDefinitionId='string'
    )
    
  :type DeviceDefinitionId: string
  :param DeviceDefinitionId: **[REQUIRED]** The ID of the device definition.

  
  
  :rtype: dict
  :returns:
