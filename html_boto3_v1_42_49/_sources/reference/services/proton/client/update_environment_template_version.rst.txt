update_environment_template_version

.. py:method:: Proton.Client.update_environment_template_version(**kwargs)

  .. danger::

        minorVersion='string',
        status='REGISTRATION_IN_PROGRESS'|'REGISTRATION_FAILED'|'DRAFT'|'PUBLISHED',
        templateName='string'
